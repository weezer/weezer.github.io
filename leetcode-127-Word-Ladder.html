import Queue, collections
class Solution(object):
    def ladderLength(self, beginWord, endWord, wordList):
        """
        :type beginWord: str
        :type endWord: str
        :type wordList: List[str]
        :rtype: int
        """
        q = Queue.Queue()
        step = 1
        q.put([beginWord, step])
        char_list = [chr(x) for x in range(ord('a'), ord('z')+1)]
        while not q.empty():
            current = q.get()
            # print current
            # print wordSet
            if current[0] == endWord:
                return current[1]
            word_length = len(current[0])
            for i in xrange(word_length):
                part1 = current[0][:i]
                part2 = current[0][i+1:]
                for j in char_list:
                    word = part1 + j + part2
                    if word in wordList:
                        q.put([word, current[1] + 1])
                        wordList.remove(word)
        return 0


    def ladderLength2(self, start, end, wordList):
        wordSet = set([])
        for word in wordList:
            wordSet.add(word)

        wordLen = len(start)
        queue = collections.deque([(start, 1)])
        while queue:
            currWord, currLen = queue.popleft()
            if currWord == end:
                return currLen
            for i in xrange(wordLen):
                part1 = currWord[:i]
                part2 = currWord[i+1:]
                for j in 'abcdefghijklmnopqrstuvwxyz':
                    if currWord[i] != j:
                        nextWord = part1 + j + part2
                        if nextWord in wordSet:
                            queue.append((nextWord, currLen + 1))
                            wordSet.remove(nextWord)
        return 0


    def ladderLength_fail(self, beginWord, endWord, wordList):
        """
        :type beginWord: str
        :type endWord: str
        :type wordList: List[str]
        :rtype: int
        """
        if endWord not in wordList:
            return 0
        self.min_count = None
        self.end_word = endWord
        copy_wordList = wordList[:]
        # for i in wordList:
        #     if self.oneLetterDiff(endWord, i) == len(endWord) and self.oneLetterDiff(beginWord, i) == len(beginWord):
        #         copy_wordList.remove(i)
        for i in copy_wordList:
            if self.oneLetterDiff(beginWord, i) == 1:
                # print beginWord, i
                copy_list = copy_wordList[:]
                copy_list.remove(i)
                self.check_word(i, 2, copy_list)
        return [0, self.min_count][self.min_count is not None]

    def check_word(self, beginWord, step, wordlist):
        if self.min_count is not None and step > self.min_count:
            return
        if beginWord == self.end_word:
            if self.min_count is None:
                self.min_count = step
            else:
                self.min_count = min(step, self.min_count)
        for i in wordlist:
            if self.oneLetterDiff(beginWord, i) == 1:
                wordlist_copy = wordlist[:]
                wordlist_copy.remove(i)
                self.check_word(i, step+1, wordlist_copy)

    def oneLetterDiff(self, firstWord, secondWord):
        word_dict = zip(firstWord, secondWord)
        count = 0
        for i, j in word_dict:
            if i != j:
                count += 1
        return count
